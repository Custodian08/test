
#include <iostream>
#include <stdlib.h>
#include <time.h>
using namespace std;

int main() {
    srand((unsigned)time(NULL));
    setlocale(LC_ALL, "Russian");
    int sum_main_diag = 0, sum_collateral_diag = 0;
    int max_sum;
    int temp;
    int index;
    int N;
    int** matrix;
    cout << "Введите размерность матрицы N = ";
    cin >> N;
    if (N <= 0) {
        cout << "Ошибочная размерность \n";
        system("pause");
        return 1;
    }
    matrix = new int* [N];
    for (int i = 0; i < N; i++)
        matrix[i] = new int[N];
    index = 0;
    max_sum = 0;
    cout << "Сформированная матрица: \n";
    for (int i = 0; i < N; i++) {
        temp = 0;
        for (int j = 0; j < N; j++) {
            matrix[i][j] = rand() % 21 - 10;
            cout << matrix[i][j] << "\t";
            if (i == j)
                sum_main_diag += matrix[i][j];
            if (i == (N - 1) - j) {
                sum_collateral_diag += matrix[i][j];
            }
        }
        cout  << endl;
        if (i == 0)
            max_sum = temp;
        else if (max_sum < temp) {
            max_sum = temp;
            index = i;
        }
    }
    cout << "Сумма главной диагонали: " << sum_main_diag << endl;
    cout << "Сумма побочной: " << sum_collateral_diag << endl;
    for (int i = 0; i < N; i++)
        delete[]matrix[i];
    delete[]matrix;
    system("pause");
    return 0;
}
